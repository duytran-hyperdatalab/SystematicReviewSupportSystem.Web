export const paperService = {
  async getUniquePapers(params: GetUniquePapersParams): Promise<GetUniquePapersResponse> {
    const queryParams = new URLSearchParams();
    if (params.search) queryParams.set("search", params.search);
    if (params.year) queryParams.set("year", params.year.toString());
    if (params.pageNumber) queryParams.set("pageNumber", params.pageNumber.toString());
    
    const url = `/identification-processes/${params.identificationProcessId}/unique-papers${queryString}`;

    return response.data;

};

### **12. Screening Resolution Architecture** 

**Key Finding:** Paper screening status is NOT stored on papers directly.

From docs and code:
```
Paper entity: title, authors, abstract, doi, year, ...
                ↓ (references)
ImportBatch: papers grouped by import
                ↓
IdentificationProcess: deduplication → unique papers
                ↓
StudySelectionProcess: papers → ScreeningResolution
                ↓
ScreeningResolution: { paperId, decision: "included"|"excluded", stage, notes }
```

This means:
- Papers are **immutable** after import
- Screening decisions are **separate entities**
- You'll need a hook like `useScreeningResolutions()` to fetch decisions
- Papers in library tab show current selection status dynamically via lookup
